Oct 4, 2014

For nothing. Don’t use for in JavaScript. Please use the .forEach array method or, use lodash’s .each functionality. It’s prettier and better!

var numbers = [1, 2, 3, 4, 5]

// SO NICE! DO THIS.
// =================
numbers.forEach(function (number) {
  console.log(number)
})

// EW GROSS. DON'T EVEN.
// ====================
for (var i = 0; i < numbers.length; i++) {
  console.log(numbers[i])
}
